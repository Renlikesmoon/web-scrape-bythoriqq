# Install
`npm i scrape-websitee`

# Usage
`const scrape = require("scrape-websitee")`

# Endpoints
- igdl2
`scrape.igdl2(url)`

- Threads
`scrape.threads(url)`

- getCerpenHorror
`scrape.getCerpenHorror()`

- fbdl
`scrape.fbdl(url)`

- twitterdl
`scrape.twitterdl(url)`

- tiktokdl
`scrape.tiktokdl(url)`

- igdl
`scrape.igdl(url)`

- cekResi
`scrape.cekResi(kurir, resi)`

- similarBand
`scrape.similarBand(query)`

- pindl
`scrape.pindl(url)`

- igStalk
`scrape.igStalk(username)`

- getCerpen
`scrape.getCerpen()`

- filmApikS
`scrape.filmApikS(query)`

- filmApikDl
`scrape.filmApikDl(url)`

- otakuDesuSearch
`scrape.otakuDesuSearch(query)`
