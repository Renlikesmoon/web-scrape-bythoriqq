# Install
`npm i scrape-website`

# Usage
`const scrape = require("scrape-website")`

# Endpoints
- tiktokdl
`scrape.tiktokdl(url)`

- igdl
`scrape.igdl(url)`

- cekResi
`scrape.cekResi(kurir, resi)`

- similarBand
`scrape.similarBand(query)`

- pindl
`scrape.pindl(url)`

- igStalk
`scrape.igStalk(username)`

- getCerpen
`scrape.getCerpen()`

- filmApikS
`scrape.filmApikS(query)`

- filmApikDl
`scrape.filmApikDl(url)

- otakuDesuSearch
`scrape.otakuDesuSearch(query)`
